2    import pandas as pd
3    import numpy as np
4    from sklearn.model_selection import train_test_split
5    from matplotlib import pyplot as plt
6    %matplotlib inline
7    from keras.models import Sequential
8    from keras.layers import Dense
9    #%% 
10   focus_columns = ['level_duration',
11                    'ob_id_ramp', 'ob_id_lever', 'ob_id_spgbrd', 'ob_id_pendulum',
12                    'ob_id_weight', 'ob_comp_freeform', 'ob_comp_pin',
13                    'CompDiff', 'pre_total']
14   
15   data = pd.read_csv("~/Downloads/updated_BIG_data_1.csv")[focus_columns].dropna() # Drop all NA values and select focus_columns
16   
17   predictors = data.drop(columns=['level_duration'])
18   num_columns = predictors.shape[1] # for modeling building
19   
20   response = np.log(data.level_duration).replace([np.inf, -np.inf], np.nan).fillna(0).round(0)
21   response.describe()
22   
23   #%% 
24   num_columns = predictors.shape[1]
25   print(f'There are {data.shape[0]} rows and {data.shape[1]} columns.')
26   print(f"The predicted variable, level_duration, is column index: {data.columns.get_loc('level_duration')}")
27   #%% 
28   X_train, X_test, y_train, y_test = train_test_split(predictors, response, test_size=0.2)
29   num_columns = X_train.shape[1]
30   
31   print("For X: Train:", X_train.shape,"Test: ", X_test.shape)
32   print("For Y: Train:", y_train.shape,"Test: ", y_test.shape)
33   #%% md 
34   ## Building and Fit the Model 
35   #%% 
36   model = Sequential()
37   model.add(Dense(12, input_dim=num_columns, activation='relu', name='input'))
38   model.add(Dense(6, activation='relu', name='hidden_6'))
39   model.add(Dense(3, activation='relu', name='hidden_3'))
40   model.add(Dense(1, activation='linear', name='exit'))
41   
42   
43   #Compile
44   model.compile(loss='mse',
45                 optimizer='adam',
46                 metrics = ['accuracy'])
47   #Fit model
48   model.fit(X_train, y_train, epochs=10, batch_size=20)
49   #%% 
50   score = model.evaluate(X_test, y_test, verbose = 0)
51   print('Test loss:', score[0])
52   print('Test Accuracy:', score[1])
53   #%% 
54   print(model.summary())
55   #%% 
56   from sklearn.metrics import confusion_matrix
57   from seaborn import heatmap
58   
59   y_predicted = model.predict(X_test).round(0)
60   conf_mat = confusion_matrix(y_test, y_predicted)
61   heatmap(conf_mat, square=True, annot=True, cmap='Blues', fmt='d', cbar=False, )
